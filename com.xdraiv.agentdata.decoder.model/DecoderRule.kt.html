package com.xdraiv.agentdata.decoder.model

import jakarta.persistence.*

@Entity
@Table(name = "client_agent_decoder")
data class DecoderRule(
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    val id: Long = 0,

    @Column(name = "client_id", nullable = false)
    val clientId: String,

    @Column(name = "agent_id", nullable = false)
    val agentId: String,

    @Column(name = "rule_name", nullable = false)
    val ruleName: String,

    @Column(name = "rule_level", nullable = false)
    var ruleLevel: Int,

    @Column(name = "rule_pattern", nullable = false)
    var rulePattern: String
)
