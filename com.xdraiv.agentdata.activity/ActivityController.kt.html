package com.xdraiv.agentdata.activity

import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.PathVariable
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController

@RestController("activity")
class ActivityController(private val repository: ActivityRepository) {

    @GetMapping("activity/event")
    fun countConnectedAgents(): List<Pair<String, Int>> {
        return repository.findAggregatedEvents()
    }

    @GetMapping("activity/all-stats")
    fun allEventStats() = repository.findAllEvents()

    @GetMapping("activity/all-events-count")
    fun allEventCount(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ) = repository.findAggregatedEventsCount(startTime, endTime)

    @GetMapping("activity/find/{eventId}")
    fun eventDetails(@PathVariable eventId: String) = repository.findByEventId(eventId)

    @GetMapping("activity/aggregate-events")
    fun dynamicAggregatedEvents(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): List<Pair<String, Int>> {
        return repository.findAggregatedEventsDynamic(startTime, endTime)
    }
}
