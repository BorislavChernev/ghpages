package com.xdraiv.agentdata.activity

import com.jillesvangurp.jsondsl.JsonDsl
import com.jillesvangurp.ktsearch.*
import com.jillesvangurp.searchdsls.querydsl.*
import com.xdraiv.agentdata.config.AppConfig
import io.ktor.client.*
import io.ktor.client.engine.cio.*
import kotlinx.coroutines.runBlocking
import kotlinx.serialization.json.*
import org.springframework.stereotype.Repository
import java.time.Instant
import java.time.LocalDate
import java.time.ZoneId
import java.time.format.DateTimeFormatter
import java.time.temporal.ChronoUnit
import java.time.temporal.WeekFields
import java.util.*
import kotlin.reflect.KProperty

class XdraivDateHistogramAggConfig : JsonDsl() {
    var field by property<String>()
    var interval by property<String>("interval")
}

class XdraivDateHistogramAgg(field: String, block: (XdraivDateHistogramAggConfig.() -> Unit)? = null) :
    AggQuery("date_histogram") {

    constructor(field: KProperty<*>, block: (XdraivDateHistogramAggConfig.() -> Unit)? = null) : this(field.name, block)

    init {
        val config = XdraivDateHistogramAggConfig()
        config.field = field
        block?.invoke(config)
        put(name, config)
    }
}


@Repository
class ActivityRepository(private val config: AppConfig) {

    private val httpClient = HttpClient(CIO) {
    }

    private val client = SearchClient(
        KtorRestClient(
            https = false,
            nodes = arrayOf(
                Node(
                    config.agentData.host,
                    config.agentData.port

                )
            ),
            client = httpClient
        )
    )

    fun getCurrentYearAndWeek(): String {
        val now = LocalDate.now()
        val weekFields = WeekFields.of(Locale.getDefault())
        val weekNumber = now.get(weekFields.weekOfWeekBasedYear())
        val year = now.year
        return "$year*"
    }

    fun findByEventId(id: String): List<ActivityLogs> = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        val response = client.search(index) {
            query = bool {
                must(
                    term("id", id)
                )
            }
        }

        response.hits!!.hits.map { Json.decodeFromString<ActivityLogs>(it.source.toString()) }
    }

    fun findAllEvents(): List<ActivityLogs> = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        val response = client.search(index) {
            query = matchAll()
        }

        response.hits!!.hits.map { Json.decodeFromString<ActivityLogs>(it.source.toString()) }
    }

    fun findAggregatedEvents(): List<Pair<String, Int>> = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        val response = client.search(index) {
            resultSize = 0
            query = bool {
                filter(
                    range("timestamp") {
                        gte = "now-24h"
                    }
                )
            }
            agg("items_per_5m", XdraivDateHistogramAgg("timestamp") {
                interval = "30m"
            })
        }

        val itemsPer5m = response.aggregations?.getAggResult<JsonObject>("items_per_5m")
        val buckets = itemsPer5m?.get("buckets") as? JsonArray

        buckets?.map { bucket ->
            val keyAsString = bucket.jsonObject["key_as_string"]?.jsonPrimitive?.content ?: ""
            val docCount = bucket.jsonObject["doc_count"]?.jsonPrimitive?.int ?: 0
            Pair(keyAsString, docCount)
        } ?: emptyList()
    }

    fun findAggregatedEventsDynamic(
        startTime: String? = null,
        endTime: String? = null
    ): List<Pair<String, Int>> = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        val formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss").withZone(ZoneId.of("UTC"))

        // If startTime or endTime is null, set default values to the last 24 hours
        val now = Instant.now()
        val defaultStartTime = formatter.format(now.minus(24, ChronoUnit.HOURS))
        val defaultEndTime = formatter.format(now)

        val start = Instant.from(formatter.parse(startTime ?: defaultStartTime))
        val end = Instant.from(formatter.parse(endTime ?: defaultEndTime))
        val durationMillis = ChronoUnit.MILLIS.between(start, end)

        // Define a target number of buckets
        val targetBuckets = 100

        // Calculate the interval dynamically
        val intervalMillis = durationMillis / targetBuckets
        val interval = when {
            intervalMillis < ChronoUnit.MINUTES.duration.toMillis() -> "1s"
            intervalMillis < ChronoUnit.HOURS.duration.toMillis() -> "${intervalMillis / ChronoUnit.MINUTES.duration.toMillis()}m"
            intervalMillis < ChronoUnit.DAYS.duration.toMillis() -> "${intervalMillis / ChronoUnit.HOURS.duration.toMillis()}h"
            else -> "${intervalMillis / ChronoUnit.DAYS.duration.toMillis()}d"
        }

        val response = client.search(index) {
            resultSize = 0
            query = bool {
                filter(
                    range("timestamp") {
                        gte = formatter.format(start)
                        lte = formatter.format(end)
                        format = "yyyy-MM-dd'T'HH:mm:ss"
                    }
                )
            }
            agg("items_per_interval", XdraivDateHistogramAgg("timestamp") {
                this.interval = interval
            })
        }

        val itemsPerInterval = response.aggregations?.getAggResult<JsonObject>("items_per_interval")
        val buckets = itemsPerInterval?.get("buckets") as? JsonArray

        buckets?.map { bucket ->
            val keyAsString = bucket.jsonObject["key_as_string"]?.jsonPrimitive?.content ?: ""
            val docCount = bucket.jsonObject["doc_count"]?.jsonPrimitive?.int ?: 0
            Pair(keyAsString, docCount)
        } ?: emptyList()
    }

    fun findAggregatedEventsCount(
        startTime: String? = null,
        endTime: String? = null
    ): Int = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        val formatter = DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss").withZone(ZoneId.of("UTC+3"))

        // If startTime or endTime is null, set default values to the last 24 hours
        val now = Instant.now()
        val defaultStartTime = formatter.format(now.minus(24, ChronoUnit.HOURS))
        val defaultEndTime = formatter.format(now)

        val start = Instant.from(formatter.parse(startTime ?: defaultStartTime))
        val end = Instant.from(formatter.parse(endTime ?: defaultEndTime))

        val response = client.search(index) {
            resultSize = 0
            query = bool {
                filter(
                    range("@timestamp") {
                        gte = formatter.format(start)
                        lte = formatter.format(end)
                        format = "yyyy-MM-dd'T'HH:mm:ss"
                    }
                )
            }
            agg("total_count", TermsAgg("_index"))
        }

        val totalCountAgg = response.aggregations?.termsResult("total_count")
        val totalCount = totalCountAgg?.buckets?.sumOf { it.jsonObject["doc_count"]?.jsonPrimitive?.int ?: 0 } ?: 0

        totalCount
    }
}
