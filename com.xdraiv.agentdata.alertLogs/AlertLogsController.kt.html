package com.xdraiv.agentdata.alertLogs

import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("/alertLogs")
class AlertLogsController(private val repository: AlertLogsRepository) {
    @GetMapping("/search")
    fun findAllAlertLogs(
        @RequestParam(defaultValue = "1") page: Int,
        @RequestParam(defaultValue = "20") size: Int,
        @RequestParam(required = false) location: String?,
        @RequestParam(required = false) fields: List<String>?,
        @RequestParam(required = false) values: List<String>?,
        @RequestParam(required = false) operations: List<String>?,
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?,
        @RequestParam(required = false) groups: List<String>?,

        ): PaginatedResponse<AlertLogs> {
        return repository.findAllAlertLogs(page, size, fields, values, operations, startTime, endTime, location, groups)
    }

    @GetMapping("/locationCounts")
    fun getLocationCounts(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Map<String, Int> {
        return repository.aggregateEventsByLocation(startTime, endTime)
    }

    @GetMapping("/AgentsCounts")
    fun getAgentsCounts(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Map<String, Int> {
        return repository.aggregateEventsByAgent(startTime, endTime)
    }

    @GetMapping("/ActionsCounts")
    fun getActionsCounts(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Map<String, Int> {
        return repository.aggregateEventsByAction(startTime, endTime)
    }

    @GetMapping("/countHighLevelAlerts")
    fun countHighLevelAlerts(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Int {
        return repository.countHighLevelAlerts(startTime, endTime)
    }

    @GetMapping("/countEventsByLocation")
    fun countEventsByLocation(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Map<String, Int> {
        return repository.countEventsByLocation(startTime, endTime)
    }

    @GetMapping("/countDistinctLocations")
    fun countDistinctLocations(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Int {
        return repository.countDistinctLocations(startTime, endTime)
    }

    @GetMapping("/alerts")
    fun countAlerts(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Int {
        return repository.countAlerts(startTime, endTime)
    }

    @GetMapping("/clusters")
    fun countCluster(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Map<String, Int> {
        return repository.countEventsByClusters(startTime, endTime)
    }

    @GetMapping("/vulnerability")
    fun countVulnerability(
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?
    ): Map<String, Int> {
        return repository.countEventsByVulnerability(startTime, endTime)
    }
}
