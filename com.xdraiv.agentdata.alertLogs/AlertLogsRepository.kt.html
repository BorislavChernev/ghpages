    fun parseDate(dateString: String): ZonedDateTime? {
        val formats = listOf(
            DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss.SSSXXX"),
            DateTimeFormatter.ofPattern("yyyy-MM-dd'T'HH:mm:ss.SSSZ")
        )
        return formats.asSequence().mapNotNull { format ->
            try {
                ZonedDateTime.parse(dateString, format)
            } catch (e: DateTimeParseException) {
                null
            }
        }.firstOrNull()
    }

    fun toIso8601Format(dateTime: String): String {
        return parseDate(dateTime)?.toInstant()?.toString() ?: dateTime
    }

    fun getCurrentYearAndWeek(): String {
        val now = LocalDate.now()
        val weekFields = WeekFields.of(Locale.getDefault())
        val weekNumber = now.get(weekFields.weekOfWeekBasedYear())
        val year = now.year
        return "$year*"
    }

    fun findAllAlertLogs(
        page: Int,
        size: Int,
        fields: List<String>?,
        values: List<String>?,
        operations: List<String>? = listOf("equals"),
        startTime: String? = null,
        endTime: String? = null,
        location: String?,
        groups: List<String>?
    ): PaginatedResponse<AlertLogs> = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        // Perform the search query and get the total number of matching documents
        val response = client.search(index) {
            resultSize = size
            from = (page - 1) * size
            query = bool {
                must(matchAll())

                fields?.forEachIndexed { index, field ->
                    val value = values?.get(index)
                    val operation = operations?.getOrNull(index) ?: "equals"
                    when (operation) {
                        "equals" -> must(term(field, value!!))
                        "gte" -> must(range(field) { gte = value!! })
                        "lte" -> must(range(field) { lte = value!! })
                        else -> must(term(field, value!!))
                    }
                }

                if (startTime != null && endTime != null) {
                    must(range("timestamp") {
                        gte = toIso8601Format(startTime)
                        lte = toIso8601Format(endTime)
                    })
                }
            }
            sort {
                add("timestamp", SortOrder.DESC)
            }
            filterSource(
                "cluster",
                "agent",
                "manager",
                "data",
                "timestamp",
                "rule.description",
                "rule.level",
                "location",
                "rule.groups"
            )
        }

        // total items count
        val totalItems = response.total

        // Calculate total pages
        val totalPages =
            if (totalItems == 0L) 1 else (totalItems / size).toInt() + if (totalItems % size == 0L) 0 else 1

        // deserialize hits to AlertLogs
        val items = response.hits?.hits?.map { json.decodeFromString<AlertLogs>(it.source.toString()) } ?: emptyList()

        PaginatedResponse(
            items = items,
            totalItems = totalItems,
            totalPages = totalPages,
            currentPage = page
        )
    }

    fun aggregateAlertLogs(
        startTime: String,
        endTime: String,
        field: String,
        page: Int,
        size: Int
    ): Map<String, Any?> = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        val isoStartTime = toIso8601Format(startTime)
        val isoEndTime = toIso8601Format(endTime)

        println("ISO Start Time: $isoStartTime")
        println("ISO End Time: $isoEndTime")

        val response = client.search(index) {
            query = bool {
                filter(
                    range("timestamp") {
                        gte = isoStartTime
                        lte = isoEndTime
                    }
                )
            }
            resultSize = size
            from = (page - 1) * size
            agg(field, TermsAgg(field) {
                aggSize = size.toLong()
            })
        }

        val buckets = response.aggregations?.termsResult(field)?.parsedBuckets
        println("BUCKETS HAH: $buckets")

        mapOf(
            "total" to response.hits?.total?.value,
            "hits" to response.hits?.hits?.map { json.decodeFromString<AlertLogs>(it.source.toString()) },
            "aggregations" to buckets
        )
    }

    @Serializable
    data class TermsBucket(
        val key: String,
        @SerialName("doc_count")
        val docCount: Long,
        @SerialName("key_as_string")
        val keyAsString: String? = null
    )

    fun aggregateEventsByLocation(
        startTime: String? = null,
        endTime: String? = null
    ): Map<String, Int> = runBlocking {
        val index = "${config.agentData.alertIndex}${getCurrentYearAndWeek()}"

        val endTimeFormatted = endTime ?: ZonedDateTime.now().format(DateTimeFormatter.ISO_INSTANT)
        val startTimeFormatted =
            startTime ?: ZonedDateTime.now().minus(24, ChronoUnit.HOURS).format(DateTimeFormatter.ISO_INSTANT)

        val response = client.search(index) {
            query = bool {
                filter(
                    range("timestamp") {
                        gte = startTimeFormatted
                        lte = endTimeFormatted
                    }
                )
            }
            resultSize = 0
            agg(
                "location_counts",
                TermsAgg("location") {
                    aggSize = 6
                }
            )
        }
        println("Response: $response")
        println("Aggregations: ${response.aggregations}")

        val buckets = response.aggregations?.get("location_counts")?.jsonObject?.get("buckets")?.jsonArray
        val result = mutableMapOf<String, Int>()

        if (buckets.isNullOrEmpty()) {
            println("No buckets found.")
        } else {
            buckets.forEach { bucket ->
                val termsBucket = json.decodeFromJsonElement<TermsBucket>(bucket)
                result[termsBucket.key] = termsBucket.docCount.toInt()
            }
        }

        result
    }
