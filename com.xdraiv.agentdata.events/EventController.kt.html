package com.xdraiv.agentdata.events

import com.xdraiv.agentdata.alertLogs.PaginatedResponse
import kotlinx.coroutines.runBlocking
import org.springframework.web.bind.annotation.GetMapping
import org.springframework.web.bind.annotation.RequestMapping
import org.springframework.web.bind.annotation.RequestParam
import org.springframework.web.bind.annotation.RestController

@RestController
@RequestMapping("/events")
class EventsController(private val eventRepository: EventRepository) {

    // Read the allowed indices from the environment variable and split them by comma
    private val allowedIndices: List<String> = System.getenv("ALLOWED_INDICES")
        ?.split(",")
        ?: listOf("events_index") // Default value if environment variable is not set

    @GetMapping("/search")
    fun findAllEvents(
        @RequestParam(defaultValue = "1") page: Int,
        @RequestParam(defaultValue = "20") size: Int,
        @RequestParam(required = false) location: String?,
        @RequestParam(required = false) fields: List<String>?,
        @RequestParam(required = false) values: List<String>?,
        @RequestParam(required = false) operations: List<String>?,
        @RequestParam(required = false) startTime: String?,
        @RequestParam(required = false) endTime: String?,
        @RequestParam(required = false, defaultValue = "events_index") indexName: String
    ): PaginatedResponse<Event> {
        if (indexName !in allowedIndices) {
            throw IllegalArgumentException("Invalid index name: $indexName")
        }
        return runBlocking {
            eventRepository.searchEvents(page, size, fields, values, operations, startTime, endTime, location, indexName)
        }
    }
}
