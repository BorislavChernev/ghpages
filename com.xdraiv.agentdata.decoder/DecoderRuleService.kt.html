package com.xdraiv.agentdata.decoder

import com.xdraiv.agentdata.decoder.model.DecoderRule
import com.xdraiv.agentdata.decoder.model.DecoderRuleDTO
import com.xdraiv.agentdata.decoder.DecoderRepository
import org.springframework.stereotype.Service

@Service
class DecoderService(val decoderRepository: DecoderRepository) {

    // Upsert method for creating or updating a rule
    fun upsertDecoderRule(clientId: String, agentId: String, ruleName: String, dto: DecoderRuleDTO): DecoderRule {
        // Check if the rule already exists in the repository
        val existingRule = decoderRepository.findByClientIdAndAgentIdAndRuleName(clientId, agentId, ruleName)

        // If the rule exists, update it; otherwise, create a new one
        val ruleEntity = existingRule?.apply {
            ruleLevel = dto.ruleLevel
            rulePattern = dto.rulePattern
        } ?: DecoderRule(
            clientId = clientId,
            agentId = agentId,
            ruleName = ruleName,
            ruleLevel = dto.ruleLevel,
            rulePattern = dto.rulePattern
        )

        // Save the updated or new rule
        return decoderRepository.save(ruleEntity)
    }

    // Return a list of DecoderRule entities
    fun getAllDecoderRules(clientId: String, agentId: String): List<DecoderRule> {
        return decoderRepository.findByClientIdAndAgentId(clientId, agentId)
    }

    // Return a single DecoderRule entity by clientId, agentId, and ruleName
    fun getDecoderRuleByName(clientId: String, agentId: String, ruleName: String): DecoderRule? {
        return decoderRepository.findByClientIdAndAgentIdAndRuleName(clientId, agentId, ruleName)
    }

    // Map DecoderRule entity to DecoderRuleDTO
    fun mapToDTO(entity: DecoderRule): DecoderRuleDTO {
        return DecoderRuleDTO(
            clientId = entity.clientId,
            agentId = entity.agentId,
            ruleName = entity.ruleName,
            ruleLevel = entity.ruleLevel,
            rulePattern = entity.rulePattern
        )
    }
}
