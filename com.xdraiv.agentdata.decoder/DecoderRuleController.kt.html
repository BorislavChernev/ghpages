package com.xdraiv.agentdata.decoder

import com.xdraiv.agentdata.decoder.model.DecoderRuleDTO
import com.xdraiv.agentdata.decoder.model.DecoderRule
import com.xdraiv.agentdata.decoder.DecoderRepository
import com.xdraiv.agentdata.decoder.model.CreateDecoderRuleDTO
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.*

@RestController
@RequestMapping("/client/{client_id}/agent/{agent_id}/decoder")
class DecoderController(val decoderService: DecoderService) {

    @PostMapping("/{rule_name}")
    fun upsertDecoderRule(
        @PathVariable("client_id") clientId: String,
        @PathVariable("agent_id") agentId: String,
        @PathVariable("rule_name") ruleName: String,
        @RequestBody createDecoderRuleDTO: CreateDecoderRuleDTO
    ): ResponseEntity<DecoderRuleDTO> {
        val savedEntity = decoderService.upsertDecoderRule(
            clientId, agentId, ruleName, DecoderRuleDTO(
                clientId = clientId,
                agentId = agentId,
                ruleName = ruleName,
                ruleLevel = createDecoderRuleDTO.ruleLevel,
                rulePattern = createDecoderRuleDTO.rulePattern
            )
        )
        return ResponseEntity(decoderService.mapToDTO(savedEntity), HttpStatus.CREATED)
    }

    @GetMapping
    fun getAllDecoderRules(
        @PathVariable("client_id") clientId: String,
        @PathVariable("agent_id") agentId: String
    ): ResponseEntity<List<DecoderRuleDTO>> {
        val rules = decoderService.getAllDecoderRules(clientId, agentId)

        // Check if the list is empty, return 204 No Content if true
        return if (rules.isEmpty()) {
            ResponseEntity(HttpStatus.NO_CONTENT)
        } else {
            ResponseEntity(rules.map { decoderService.mapToDTO(it) }, HttpStatus.OK)
        }
    }

    @GetMapping("/{rule_name}")
    fun getDecoderRuleByName(
        @PathVariable("client_id") clientId: String,
        @PathVariable("agent_id") agentId: String,
        @PathVariable("rule_name") ruleName: String
    ): ResponseEntity<DecoderRuleDTO> {
        val rule = decoderService.getDecoderRuleByName(clientId, agentId, ruleName)
        return if (rule != null) {
            ResponseEntity(decoderService.mapToDTO(rule), HttpStatus.OK)
        } else {
            ResponseEntity(HttpStatus.NOT_FOUND)
        }
    }
}
