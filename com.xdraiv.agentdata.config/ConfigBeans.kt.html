package com.xdraiv.agentdata.config

import com.typesafe.config.ConfigFactory
import org.springframework.context.annotation.Bean
import org.springframework.context.annotation.Configuration
import org.springframework.web.servlet.config.annotation.CorsRegistry
import org.springframework.web.servlet.config.annotation.WebMvcConfigurer

@Configuration
class ConfigBeans : WebMvcConfigurer {

    @Bean
    fun corsConfigBean(): WebMvcConfigurer {
        return object : WebMvcConfigurer {
            override fun addCorsMappings(registry: CorsRegistry) {
                registry.addMapping("/**")  // Apply to all endpoints
                    .allowedOriginPatterns("*")  // Your frontend URL
                    .allowedMethods("GET", "POST", "PUT", "DELETE", "OPTIONS")  // Allowed HTTP methods
                    .allowedHeaders("*")  // Allow all headers
                    .allowCredentials(true)  // Allow credentials like cookies or headers
            }
        }
    }

    @Bean
    fun appConfigBean(): AppConfig {
        val config = ConfigFactory.load()

        // Load the values from the config file for each parameter
        return AppConfig(
            port = config.getInt("app.port"),  // Port number
            host = config.getString("app.host"),
            googleOauthUrl = config.getString("app.google-oauth-url"),

            // Load clientDataConfig
            clientData = ClientDataConfig(
                url = config.getString("app.client-data.url"),
                user = config.getString("app.client-data.user"),
                password = config.getString("app.client-data.password")
            ),

            // Load agentDataConfig
            agentData = AgentDataConfig(
                host = config.getString("app.agent-data.host"),
                port = config.getInt("app.agent-data.port"),
                alertIndex = config.getString("app.agent-data.alert-index"),
                agentIndex = config.getString("app.agent-data.agent-index")

            )
        )
    }
}
