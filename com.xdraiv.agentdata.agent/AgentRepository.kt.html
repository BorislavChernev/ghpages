package com.xdraiv.agentdata.agent

import com.jillesvangurp.ktsearch.*
import com.jillesvangurp.searchdsls.querydsl.*
import com.xdraiv.agentdata.agent.model.Agent
import com.xdraiv.agentdata.agent.model.termsResult
import com.xdraiv.agentdata.config.AppConfig
import io.ktor.client.*
import io.ktor.client.engine.cio.*
import kotlinx.coroutines.runBlocking
import kotlinx.serialization.json.Json
import org.slf4j.LoggerFactory
import org.springframework.stereotype.Repository
import java.time.LocalDate
import java.time.temporal.WeekFields
import java.util.*

fun TopHitsAggConfig.sourceIncludes(includes: List<String>) {
	this["_source"] = mapOf("includes" to includes)
}

@Repository
class AgentRepository(val config: AppConfig) {

	private val logger = LoggerFactory.getLogger(AgentRepository::class.java)

	private val httpClient = HttpClient(CIO) {
	}

	private val client = SearchClient(
		KtorRestClient(
			https = false,
			nodes = arrayOf(
				Node(
					config.agentData.host,
					config.agentData.port
				)
			),
			client = httpClient
		)
	)

	private val json = Json {
		ignoreUnknownKeys = true
	}

	fun getCurrentYearAndWeek(): String {
		val now = LocalDate.now()
		val weekFields = WeekFields.of(Locale.getDefault())
		val weekNumber = now.get(weekFields.weekOfWeekBasedYear())
		val year = now.year
		return "$year*"
	}

	fun findByAgentId(id: String): List<Agent> = runBlocking {
		val index = "${config.agentData.agentIndex}${getCurrentYearAndWeek()}"

		val response = client.search(index) {
			query = bool {
				must(
					term("id", id)
				)
			}
		}

		response.hits!!.hits.map { json.decodeFromString<Agent>(it.source.toString()) }
	}

	fun latestRegisteredAgentName(): String? = runBlocking {
		val index = "${config.agentData.agentIndex}${getCurrentYearAndWeek()}"

		val response = client.search(index) {
			query = matchAll()
			sort {
				add("dateAdd", SortOrder.DESC)
			}
			resultSize = 1
		}

		response.hits!!.hits.firstOrNull()?.source?.let {
			json.decodeFromString<Agent>(it.toString()).name
		}
	}

	fun findAllAgents(): List<Agent> = runBlocking {
		val index = "${config.agentData.agentIndex}${getCurrentYearAndWeek()}"

		val response = client.search(index) {
			query = matchAll()
			resultSize = 1000 // Increase the result size to fetch more data
		}

		response.hits!!.hits.map { json.decodeFromString<Agent>(it.source.toString()) }
	}

	fun countAgentsByActivityStatus(): MutableMap<String, Int> = runBlocking {
		val index = "${config.agentData.agentIndex}${getCurrentYearAndWeek()}"

		val statusCount = mutableMapOf("active" to 0, "disconnected" to 0, "never_connected" to 0)

		var currentFrom = 0
		val batchSize = 10000
		var totalResults: Long

		do {
			val response = client.search(index) {
				query = bool {
					filter(range("timestamp") {
						gte = "now-1w"
					})
				}
				resultSize = batchSize
				from = currentFrom
				agg("by_agent", TermsAgg("id") {
					aggSize = batchSize.toLong()
				}) {
					agg("latest_status", TopHitsAgg {
						resultSize = 1
						sort {
							add("timestamp")
						}
						sourceIncludes(listOf("status", "timestamp"))
					})
				}
			}
			totalResults = response.hits!!.total?.value ?: 0
			val byAgent = response.aggregations?.termsResult("by_agent")

			byAgent?.parsedBuckets?.forEach { bucket ->
				val topHits = bucket.aggregations.topHitResult("latest_status")
				val status = topHits.hits.hits.firstOrNull()?.source?.get("status")?.toString()?.replace("\"", "")

				if (status != null) {
					statusCount[status] = statusCount.getOrDefault(status, 0) + 1
				}
			}

			currentFrom += batchSize
		} while (currentFrom < totalResults)

		logger.info("Status Count: $statusCount")
		statusCount
	}

	fun countAgentsByOS(): MutableMap<String, Int> = runBlocking {
		val index = "${config.agentData.agentIndex}${getCurrentYearAndWeek()}"

		val platformCount = mutableMapOf("windows" to 0, "ubuntu" to 0, "darwin" to 0)

		var currentFrom = 0
		val batchSize = 10000
		var totalResults: Long

		do {
			val response = client.search(index) {
				query = bool {
					filter(range("timestamp") {
						gte = "now-1w"
					})
				}
				resultSize = batchSize
				from = currentFrom
				agg("by_os", TermsAgg("id") {
					aggSize = batchSize.toLong()
				}) {
					agg("latest_os", TopHitsAgg {
						resultSize = 1
						sort {
							add("timestamp")
						}
						sourceIncludes(listOf("os.platform", "timestamp"))
					})
				}
			}
			totalResults = response.hits!!.total?.value ?: 0
			val byOS = response.aggregations?.termsResult("by_os")
			byOS?.parsedBuckets?.forEach { bucket ->
				val topHits = bucket.aggregations.topHitResult("latest_os")
				val sourceMap = topHits.hits.hits.firstOrNull()?.source
				val platform = (sourceMap?.get("os") as? Map<*, *>)?.get("platform")?.toString()?.replace("\"", "")
				platform?.let {
					// Normalize platform names to match the keys in platformCount
					val normalizedPlatform = when(it.lowercase()) {
						"windows" -> "windows"
						"ubuntu" -> "ubuntu"
						"darwin" -> "darwin"
						"macos" -> "macOS"
						else -> null
					}
					normalizedPlatform?.let { normalized ->
						platformCount[normalized] = platformCount.getOrDefault(normalized, 0) + 1
					}
				}
			}

			currentFrom += batchSize
		} while (currentFrom < totalResults)

		logger.info("Status Count: $platformCount")
		platformCount
	}

	fun countConnectedAgents(): Int {
		return countAgentsByActivityStatus()["active"] ?: 0
	}

	fun countDisconnectedAgents(): Int {
		return countAgentsByActivityStatus()["disconnected"] ?: 0
	}

	fun countNeverConnectedAgents(): Int {
		return countAgentsByActivityStatus()["never_connected"] ?: 0
	}

	fun listAgents(): List<String> = runBlocking {
		val index = "${config.agentData.agentIndex}${getCurrentYearAndWeek()}"

		val agentNames = mutableSetOf<String>() // Using a Set to ensure uniqueness

		var currentFrom = 0
		val batchSize = 1000
		var totalResults: Long

		do {
			val response = client.search(index) {
				query = bool {
					filter(range("timestamp") {
						gte = "now-6h"
					})
				}
				resultSize = batchSize
				from = currentFrom
			}

			totalResults = response.hits!!.total?.value ?: 0

			// Extract agent names from the hits and add them to the Set
			response.hits!!.hits.forEach { hit ->
				hit.source?.get("name")?.toString()?.replace("\"", "").let {
					if (it != null) {
						agentNames.add(it)
					}
				}
			}

			currentFrom += batchSize
		} while (currentFrom < totalResults)

		agentNames.toList()
	}
}
