package com.xdraiv.agentdata.client

import com.fasterxml.jackson.module.kotlin.jacksonObjectMapper
import org.springframework.http.HttpStatus
import org.springframework.http.ResponseEntity
import org.springframework.web.bind.annotation.*

@RestController
@RequestMapping("/client")
class ClientController(
    val clientSettingsRepository: ClientRepository,
    val authorizationService: AuthorizationService
) {

    // Fetch client settings by client_id
    @GetMapping("/{client_id}/settings")
    fun getClientSettings(@PathVariable("client_id") clientId: String): ResponseEntity<ClientResponse> {
        val client = clientSettingsRepository.findByClientId(clientId)
            ?: return ResponseEntity.status(HttpStatus.NOT_FOUND).build() // Return 404 if client not found

        val resp = ClientResponse(
            clientId = client?.clientId,
            oauthJson = jacksonObjectMapper().readValue(client?.credentialsJson, Map::class.java) as Map<String, Any>,
            refreshToken = jacksonObjectMapper().readValue(client?.refreshToken, Map::class.java) as Map<String, Any>?
        )
        return ResponseEntity.ok(resp)
    }


    @PutMapping("/{client_id}/settings")
    fun upsertClientSettings(
        @PathVariable("client_id") clientId: String,
        @RequestBody oauthJson: Map<String, Any>
    ): ResponseEntity<Void> {
        // Create a Client object with the client_id and oauthJson
        val client = Client(
            clientId = clientId,
            credentialsJson = jacksonObjectMapper().writeValueAsString(oauthJson),
            refreshToken = ""
        )

        return if (clientSettingsRepository.upsertClientSettings(client)) {
            ResponseEntity.status(HttpStatus.CREATED).build()
        } else {
            ResponseEntity.notFound().build()
        }
    }


    @PostMapping("/{client_id}/authorize/gcp-logs/")
    suspend fun authorizationCallback2GCPLogs(
        @PathVariable("client_id") clientId: String,
        @RequestBody requestBody: Map<String, Any>
    ) {
        val code = requestBody["code"]
        val state = requestBody["state"]
        // Verify 'state' to prevent CSRF
        if (code != null) {
            // Exchange code for tokens
            authorizationService.fetchClientToken(clientId)
//            call.respondText("Authorization successful. You can close this window.")
        } else {
            ResponseEntity.badRequest().build<Void>()
        }
    }
}
