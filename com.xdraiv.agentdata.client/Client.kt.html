package com.xdraiv.agentdata.client

import com.fasterxml.jackson.module.kotlin.jacksonObjectMapper
import jakarta.persistence.Column
import jakarta.persistence.Entity
import jakarta.persistence.Id
import jakarta.persistence.Table

@Entity
@Table(name = "clients", schema = "public")
data class Client(
    @Id
    @Column(name = "client_id")
    val clientId: String,

    @Column(name = "credentials_json")
    var credentialsJson: String,

    @Column(name = "refresh_token")
    var refreshToken: String,
) {
    constructor() : this("", "", "")

    fun getOauthJson(): Map<String, Any> {
        val mapper = jacksonObjectMapper()
        return mapper.readValue(credentialsJson, Map::class.java) as Map<String, Any>
    }

    fun setOauthJson(oauthJson: Map<String, Any>) {
        val mapper = jacksonObjectMapper()
        this.credentialsJson = mapper.writeValueAsString(oauthJson)
    }

    fun getRefreshToken(): Map<String, Any> {
        val mapper = jacksonObjectMapper()
        return mapper.readValue(refreshToken, Map::class.java) as Map<String, Any>
    }

    fun setRefreshToken(refreshToken: Map<String, Any>) {
        val mapper = jacksonObjectMapper()
        this.refreshToken = mapper.writeValueAsString(refreshToken)
    }
}
