package com.xdraiv.agentdata.client

import com.fasterxml.jackson.module.kotlin.jacksonObjectMapper
import jakarta.persistence.EntityManager
import jakarta.persistence.NoResultException
import jakarta.persistence.PersistenceContext
import org.springframework.stereotype.Repository
import org.springframework.transaction.annotation.Transactional

@Repository
class ClientRepository(
    @PersistenceContext private val entityManager: EntityManager
) {
    private val mapper = jacksonObjectMapper()

    fun findByClientId(clientId: String): Client? {
        val query = entityManager.createQuery(
            "SELECT c FROM Client c WHERE c.clientId = :client_id", Client::class.java
        )
        query.setParameter("client_id", clientId)
        return query.resultList.firstOrNull()

    }

    @Transactional
    fun upsertClientSettings(newClient: Client): Boolean {
        val existingClient = findByClientId(newClient.clientId)

        return runCatching {
            if (existingClient == null) {
                entityManager.persist(newClient)
            } else {
                val updatedClient = existingClient.copy(
                    credentialsJson = mapper.writeValueAsString(existingClient.getOauthJson()),
                    refreshToken = mapper.writeValueAsString(
                        existingClient.getRefreshToken().takeIf { it.isNotEmpty() }
                            ?: emptyMap<String, Any>())
                )
                entityManager.merge(updatedClient)
            }

        }.isSuccess
    }
}
