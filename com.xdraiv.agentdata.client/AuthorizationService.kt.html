package com.xdraiv.agentdata.client

import com.google.api.client.auth.oauth2.TokenResponse
import com.xdraiv.agentdata.config.AppConfig
import io.ktor.client.*
import io.ktor.client.call.*
import io.ktor.client.request.*
import io.ktor.http.*
import org.springframework.stereotype.Service

@Service
class AuthorizationService(val clientRepository: ClientRepository, val config: AppConfig) {

    suspend fun fetchClientToken(clientId: String): TokenResponse? {
        val client = clientRepository.findByClientId(clientId) ?: return null

        val httpClient = HttpClient()
        val response: TokenResponse = httpClient.post(config.googleOauthUrl) {
            contentType(ContentType.Application.FormUrlEncoded)
            Parameters.build {
                append("client_id", clientId)
                append("client_secret", client.credentialsJson)
                append("grant_type", "authorization_code")
            }
        }.body()

        val updatedClient = client.copy(
            credentialsJson = response.accessToken,
            refreshToken = response.refreshToken
        )
        clientRepository.upsertClientSettings(updatedClient)

        return response
    }
}
