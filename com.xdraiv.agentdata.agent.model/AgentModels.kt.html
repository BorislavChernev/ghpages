package com.xdraiv.agentdata.agent.model

import com.jillesvangurp.ktsearch.Aggregations
import com.jillesvangurp.ktsearch.Bucket
import com.jillesvangurp.ktsearch.getAggResult
import kotlinx.datetime.Instant
import kotlinx.serialization.SerialName
import kotlinx.serialization.Serializable
import kotlinx.serialization.json.Json
import kotlinx.serialization.json.JsonObject


@Serializable
data class OS(
    val platform: String
)

@Serializable
data class Agent(
    val id: String,
    val name: String,
    val ip: String,
    val timestamp: Instant,
    val os: OS? = null,
    val status: String
)

@Serializable
data class TermsBucket(
    val key: String,
    @SerialName("doc_count")
    val docCount: Long
)

@Serializable
data class TermsAggregationResult(
    @SerialName("doc_count_error_upper_bound")
    val docCountErrorUpperBound: Long,
    @SerialName("sum_other_doc_count")
    val sumOtherDocCount: Long,
    val buckets: List<JsonObject>
) {
    val parsedBuckets: List<Bucket<TermsBucket>>
        get() = buckets.map { Bucket(it, TermsBucket.serializer()) }
}

fun Aggregations?.termsResult(name: String, json: Json = Json): TermsAggregationResult =
    this?.getAggResult(name, json) ?: TermsAggregationResult(0, 0, emptyList())
